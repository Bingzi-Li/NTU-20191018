def main_cast(cast, n_main=3):
    cast_pattern = np.array([np.nan, None, np.nan])
    if cast.empty:
        return np.tile(cast_pattern, n_main)
    else:
        n_cast = len(cast)
        cast_extract = cast.loc[:min(n_main-1, n_cast), ['id','name', 'gender']].values.flatten()
        return np.append(cast_extract, np.tile(cast_pattern, max(0, n_main-n_cast)))
